\section{Different widths of whitespace commands.}

\begin{itemize}
    \item \itemWidth{Prof. A.\, Bauer}
    \item \itemWidth{Prof. A.\ Bauer}
    \item \itemWidth{Prof.~A.~Bauer}
    \item \itemWidth{Prof. A. Bauer.}
\end{itemize}

\emph{We should use the third option \emph{\texttt{"Prof.\~A.\~Bauer"}}}
